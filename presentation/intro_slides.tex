\begin{frame}{Main Theorem}
    \begin{theorem}
      There exists a certification algorithm that takes a classical description of an $n$-qubit {\bf pure} state $\ket{\psi_{\text{target}}}$ and a single copy of an $n$-qubit {\bf mixed} state $\rho_{\text{lab}}$, performs only {\bf single-qubit measurements}, and outputs:
        \begin{itemize}
            \item \textsc{Accept} with probability at least
            $\mathrm{Fid} := \bra{\psi_{\text{target}}}\rho_{\text{lab}}\ket{\psi_{\text{target}}}$
            \item \textsc{Reject} with probability at least
            $\mathrm{Infid}/n := (1-\mathrm{Fid})/n$
        \end{itemize}
    \end{theorem}

    \pause
    To get an arbitrarily high success rate $1 - \epsilon$, just repeat $\frac n {\mathrm{Infid}} \log(1/\epsilon)$ times.
    %\begin{center}
    %    One qubit at a time: compare locally, guarantee globally.
    %\end{center}
\end{frame}

\begin{frame}{Access Model}
  \begin{block}{Definition (Oracle)}
    Input: product operator $ \Pi=\bigotimes_{i=1}^n \Pi_i $, each $\Pi_i$ operates on only 1 bit
      \pause

      \vspace{0.5em}
      Output: $ \bra{\psi_{\text{target}}}\Pi\ket{\psi_{\text{target}}}$
    \end{block}

    \pause
    \vspace{0.5em}
        Huang et al. use a similar oracle restricted to the computational basis
\end{frame}

\begin{frame}{Using the Access Model}
          
            \begin{itemize}
              \item Iteratively build $\Pi_t$ and query
            \[
                \Pi_t \otimes \ketbra{a}{a} \otimes I^{\otimes(n-t-1)}
            \]
            for $a\in\{0,+,\mathfrak{i}\}$

            \pause
          \item Reconstruct the marginal on bit $t+1$
            \pause
          \item Query the lab state and update $\Pi_t$
          \item Repeat
            \end{itemize}
\end{frame}
