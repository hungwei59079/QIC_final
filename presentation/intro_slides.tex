\section{Introduction}

\begin{frame}{State Certification}
    \begin{center}
        \resizebox{0.78\textwidth}{!}{\input{../diagrams/huang_state_certification.tex}}
    \end{center}

    \vspace{-0.3em}
    \begin{block}{Goal}
        Given a classical description of a target pure state $\ket{\psi_{\text{target}}}$ and one copy of an unknown lab state $\rho_{\text{lab}}$, decide whether
        \[
            \mathrm{Fid}=\bra{\psi_{\text{target}}}\rho_{\text{lab}}\ket{\psi_{\text{target}}}
        \]
        is large, using only simple measurements on $\rho_{\text{lab}}$.
    \end{block}
\end{frame}

\begin{frame}{Main Guarantee}
    \begin{theorem}
        There exists a certification algorithm that takes a classical description of an $n$-qubit pure state $\ket{\psi_{\text{target}}}$ and a single copy of an $n$-qubit state $\rho_{\text{lab}}$, performs only single-qubit measurements, and outputs:
        \begin{itemize}
            \item \textsc{Accept} with probability at least
            $\mathrm{Fid} := \bra{\psi_{\text{target}}}\rho_{\text{lab}}\ket{\psi_{\text{target}}}$
            \item \textsc{Reject} with probability at least
            $\mathrm{Infid}/n := (1-\mathrm{Fid})/n$
        \end{itemize}
    \end{theorem}

    %\begin{center}
    %    One qubit at a time: compare locally, guarantee globally.
    %\end{center}
\end{frame}

\begin{frame}{Naive Certification}
    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            For a known pure target, certification can be sample-efficient if we can perform the target projector
            \[
                \left\{\ket{\psi_{\text{target}}}\!\bra{\psi_{\text{target}}},
                I-\ket{\psi_{\text{target}}}\!\bra{\psi_{\text{target}}}\right\}.
            \]

            \vspace{0.5em}
            Equivalently, one may use a SWAP test against a freshly prepared copy of $\ket{\psi_{\text{target}}}$.

\begin{frame}{Classical Shadows Help, But Have Tradeoffs}
    \begin{center}
        \resizebox{0.82\textwidth}{!}{\input{../diagrams/huang_classical_shadows.tex}}
    \end{center}

    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \begin{block}{Random Clifford shadows}
                Dimension-independent sample complexity for fidelity estimation, but require deep circuits and hard target overlaps for general $\ket{\psi}$.
            \end{block}
        \end{column}
        \pause
        \begin{column}{0.48\textwidth}
            \begin{block}{Single-qubit Pauli shadows}
                Experimentally simple, but worst-case sample complexity grows exponentially for highly entangled targets.
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Existing Simple-Measurement Protocols Are Specialized}
    \begin{itemize}
        \item Direct fidelity estimation and Pauli-shadow variants work well for structured observables and target families.
        \item Efficient certification is known for stabilizer states, shallow-circuit outputs, hypergraph states, IQP outputs, and Gaussian states.
        \item For generic random targets, XEB can work under suitable noise assumptions, but it is not a general certification guarantee.
    \end{itemize}

\begin{frame}{Motivating Applications}
    \begin{center}
        \resizebox{0.86\textwidth}{!}{\input{../diagrams/huang_applications_overview.tex}}
    \end{center}

    \begin{itemize}
        \item Benchmark a noisy quantum device against its intended output.
        \item Train or certify neural quantum-state models using fidelity as the objective.
        \item Optimize parametrized circuits toward a target state without implementing the target projector.
    \end{itemize}
\end{frame}

\begin{frame}{What We Need From the Protocol}
    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            \begin{enumerate}
                \item Use only single-qubit measurements on $\rho_{\text{lab}}$.
                \item Use the classical description of $\ket{\psi_{\text{target}}}$ only for choosing measurement bases and computing conditioned target states.
                \item Accumulate local evidence that provably controls the global fidelity.
            \end{enumerate}
        \end{column}
        \begin{column}{0.46\textwidth}
