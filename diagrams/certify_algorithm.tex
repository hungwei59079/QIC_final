\begin{tikzpicture}[>=Stealth]

% --- Qubit register ---
\def\qh{0.5}  % qubit box height
\def\qw{0.55} % qubit box width
\def\y{3.0}   % vertical position

% Label
\node[font=\footnotesize, anchor=east] at (-0.3, \y)
  {$|\mathsf{lab}\rangle$};

% qubits 1 .. k-1 (computational basis)
\foreach \i in {0,1,2} {
  \draw[fill=black!12]
    ({\i*\qw}, {\y-\qh/2})
    rectangle +(\qw, \qh);
}
\node[font=\tiny] at ({1.5*\qw}, \y) {$\cdots$};

% qubit k (measured last)
\draw[fill=white, thick]
  ({3.5*\qw}, {\y-\qh/2})
  rectangle +(\qw, \qh);
\node[font=\scriptsize] at ({4.0*\qw}, \y) {$k$};

% qubits k+1 .. n (DT basis)
\foreach \i in {5,6,7,8} {
  \draw[fill=black!5,
    pattern=north east lines,
    pattern color=black!15]
    ({\i*\qw-0.5*\qw}, {\y-\qh/2})
    rectangle +(\qw, \qh);
}
\node[font=\tiny] at ({7.0*\qw}, \y)
  {$\cdots$};

% --- Measurement arrows and outcome nodes ---

% Left: comp basis -> outcome x
\draw[->] ({1.5*\qw}, {\y-\qh/2-0.1})
  -- ({1.5*\qw}, {\y-\qh/2-0.7});
\node[font=\scriptsize, draw, rounded corners=1pt,
  inner sep=3pt] (xnode)
  at ({1.5*\qw}, {\y-\qh/2-1.05})
  {$x \in \{0,1\}^{k-1}$};

% Right: DT basis -> outcome ell
\draw[->] ({6.5*\qw}, {\y-\qh/2-0.1})
  -- ({6.5*\qw}, {\y-\qh/2-0.7});
\node[font=\scriptsize, draw, rounded corners=1pt,
  inner sep=3pt] (ellnode)
  at ({6.5*\qw}, {\y-\qh/2-1.05})
  {$\ell$ from $\mathcal{T}_x$};

% Adaptive dependence: x -> T_x
\draw[-{Stealth[length=5pt]}, dashed, semithick]
  (xnode) -- (ellnode);

% Merge: x and ell feed into final measurement
\node[draw, rounded corners=2pt,
  font=\scriptsize, inner sep=4pt] (meas)
  at ({4.0*\qw}, {\y-\qh/2-2.2})
  {Measure qubit $k$ against
   $|\mathsf{hyp}'\rangle$
   given $x,\ell$};

% Center: qubit k -> measurement box
\draw[->] ({4.0*\qw}, {\y-\qh/2-0.1}) -- (meas.north);

% x and ell feed straight down into measurement box
\draw[->, densely dashed] (xnode.south) -- (xnode.south |- meas.north);
\draw[->, densely dashed] (ellnode.south) -- (ellnode.south |- meas.north);

% Accept/Reject
\draw[->] (meas.south) -- ++(0, -0.5);
\node[font=\footnotesize] at ({4.0*\qw}, {\y-\qh/2-3.1})
  {\textsc{Accept} / \textsc{Reject}};

\end{tikzpicture}
