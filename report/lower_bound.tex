\section{Lower Bound}
We aim to construct states $\ket{\varphi_C}$ and $\rho_C$ such that their measurement outcomes are close, while the fidelity is small enough so that the algorithm will reject.

\subsection{SIC-POVM states}
A POVM is said to be \textit{informationally complete} if the quantum state can be uniquely determined from the measurement outcome probabilities.
Since the real vector space of Hermitian matrices on $\mathbb{C}^d$ has real dimension $d^2$, any such POVM must have at least $d^2$ elements.
If we further require that the pairwise Hilbert--Schmidt inner products between distinct POVM elements are equal, we call such a POVM a SIC-POVM.
Equivalently, a SIC-POVM in $\mathbb{C}^d$ can be specified by $d^2$ pure states $\ket{\chi_{k}}$ such that 
\begin{align}\label{eq_202606101020}
\left|\bra{\chi_{i}}\ket{\chi_j}\right|^2=\frac{1}{d+1},\quad i\neq j.
\end{align}
The corresponding POVM element is
\begin{align}\label{eq_202606101437}
E_k=\frac{1}{d}\ket{\chi_k}\bra{\chi_k}.
\end{align}
\begin{remark}
It is still unknown whether SIC-POVMs exist in every dimension $d$~\cite{10.1063/1.1737053}.
\end{remark}
Luckily, in the qubit case $d=2$, we can construct a SIC-POVM as follows.
Let
\begin{align*}
r_1&=\frac{1}{\sqrt{3}}(1,1,1),\quad r_2=\frac{1}{\sqrt{3}}(1,-1,-1),\quad r_3=\frac{1}{\sqrt{3}}(-1,1,-1),\quad r_4=\frac{1}{\sqrt{3}}(-1,-1,1).
\end{align*}
Define the rank-one projection
\begin{align}\label{eq_202606101024}
\ket{\chi_k}\bra{\chi_k}=\frac{1}{2}\left(I_2+r_k\cdot\sigma\right)
\end{align}
where $\sigma$ is the Pauli vector.
Then, it is a valid POVM, since
\begin{align*}
\sum_{k=1}^4 E_k=I_2+\frac{1}{4}\left(\sum_{k=1}^4r_k\right)\cdot\sigma=I_2.
\end{align*}

To check \eqref{eq_202606101020}, we compute
\begin{align}\label{eq_202606101002}
\left|\bra{\chi_{i}}\ket{\chi_j}\right|^2&=\tr\left(\ket{\chi_i}\bra{\chi_i}\ket{\chi_j}\bra{\chi_j}\right)=\frac{1}{4}\tr\left(I_2+r_i\cdot\sigma+r_j\cdot\sigma+\left(r_i\cdot\sigma\right)\left(r_j\cdot\sigma\right)\right).
\end{align}
Note that 
\begin{align*}
\tr\left(r_i\cdot\sigma\right)=\sum_{\ell=1}^3r_{i\ell}\tr\sigma_{\ell}=0
\end{align*}
and
\begin{align*}
\tr\left(\left(r_i\cdot\sigma\right)\left(r_j\cdot\sigma\right)\right)&=\sum_{a,b=1}^3r_{ia}r_{jb}\tr\left(\sigma_a\sigma_b\right)=2\sum_{a,b=1}^3r_{ia}r_{jb}\delta_{ab}=2r_i\cdot r_j.
\end{align*}
Also, by computation,
\begin{align*}
r_i\cdot r_j=-\frac{1}{3},\quad i\neq j.
\end{align*}
Thus, \eqref{eq_202606101002} becomes
\begin{align*}
\left|\bra{\chi_{i}}\ket{\chi_j}\right|^2&=\frac{1}{2}\left(1+r_i\cdot r_j\right)=\frac{1}{3}=\frac{1}{2+1}.
\end{align*}
Hence, $\{E_k\}_{k=1}^4$ represents a valid SIC-POVM in dimension $2$.

We now show that, for any one-qubit measurement basis, at least one of the cross terms $\ket{\chi_1}\bra{\chi_b}$ is suppressed for some $b\in\{2,3,4\}$.
\begin{lemma}
For any orthonormal basis $\{\ket{\varphi},\ket{\varphi^\perp}\}$ of $\mathbb{C}^2$, there exists $b\in\{2,3,4\}$ such that
\begin{align}\label{eq_202606101458}
\left|\bra{\varphi}\ket{\chi_1}\bra{\chi_b}\ket{\varphi}\right|+\left|\bra{\varphi^\perp}\ket{\chi_1}\bra{\chi_b}\ket{\varphi^\perp}\right|\le\sqrt{\frac{8}{9}}.
\end{align}
\end{lemma}
\begin{proof}
Let the Bloch vector of $\ket{\varphi}$ be $v\in\mathbb{R}^3$.
Then, by orthogonality, the Bloch vector of $\ket{\varphi^\perp}$ is $-v$.
As a result, we have
\begin{align*}
\left|\bra{\varphi}\ket{\chi_k}\right|^2&=\tr\left(\ket{\varphi}\bra{\varphi}\ket{\chi_k}\bra{\chi_k}\right)=\frac{1}{4}\tr\left(I_2+v\cdot\sigma+r_k\cdot\sigma+\left(v\cdot\sigma\right)\left(r_k\cdot\sigma\right)\right)=\frac{1}{2}\left(1+v\cdot r_k\right)
\end{align*}
and
\begin{align*}
\left|\bra{\varphi^\perp}\ket{\chi_k}\right|^2&=\tr\left(\ket{\varphi^\perp}\bra{\varphi^\perp}\ket{\chi_k}\bra{\chi_k}\right)=\frac{1}{4}\tr\left(I_2-v\cdot\sigma+r_k\cdot\sigma-\left(v\cdot\sigma\right)\left(r_k\cdot\sigma\right)\right)=\frac{1}{2}\left(1-v\cdot r_k\right)
\end{align*}
for any $k$.
Let $\alpha=v\cdot r_1$ and $\beta=v\cdot r_k$.
Then, \eqref{eq_202606101458} is equivalent to
\begin{align*}
\left(\frac{1}{2}\sqrt{(1+\alpha)(1+\beta)}+\frac{1}{2}\sqrt{(1-\alpha)(1-\beta)}\right)^2=\frac{1}{2}\left(1+\alpha\beta+\sqrt{\left(1-\alpha^2\right)\left(1-\beta^2\right)}\right)\le\frac{8}{9}.
\end{align*}

By direct computations,
\begin{align*}
v\cdot\left(r_1-r_2\right)&=\frac{2}{\sqrt{3}}\left(v_2+v_3\right)\\
v\cdot\left(r_1-r_3\right)&=\frac{2}{\sqrt{3}}\left(v_1+v_3\right)\\
v\cdot\left(r_1-r_4\right)&=\frac{2}{\sqrt{3}}\left(v_1+v_2\right).
\end{align*}
Also,
\begin{align*}
\left(v_1+v_2\right)^2+\left(v_2+v_3\right)^2+\left(v_1+v_3\right)^2=1+\left(v_1+v_2+v_3\right)^2\ge1.
\end{align*}
Thus, there exists $b\in\{2,3,4\}$ such that
\begin{align*}
|\alpha-\beta|=\left|v\cdot\left(r_1-r_b\right)\right|\ge\frac{2}{\sqrt{3}}\cdot\frac{1}{\sqrt{3}}=\frac{2}{3}.
\end{align*}
Observe that
\begin{align*}
\sqrt{\left(1-\alpha^2\right)\left(1-\beta^2\right)}\le\frac{2-\alpha^2-\beta^2}{2}.
\end{align*}
Hence,
\begin{align*}
\frac{1}{2}\left(1+\alpha\beta+\sqrt{\left(1-\alpha^2\right)\left(1-\beta^2\right)}\right)\le\frac{1}{2}\left(2-\frac{1}{2}\left(\alpha-\beta\right)^2\right)\le\frac{1}{2}\left(2-\frac{1}{2}\cdot\frac{4}{9}\right)=\frac{8}{9}.
\end{align*}
This proves the lemma.
\end{proof}

\subsection{The hard-to-certify state}
Let $\ket{\chi_1},\ket{\chi_2},\ket{\chi_3},\ket{\chi_4}\in\mathbb{C}^2$ be pure states corresponding to a SIC-POVM.
Pick any $N\in\mathbb{N}$ and set $C=(c^1,\cdots,c^N)$ where $c^k\in\{1,2,3,4\}^n$.
We denote
\begin{align*}
\ket{\psi_{c^t}}=\ket{\chi_{c^t_1}}\otimes\cdots\otimes\ket{\chi_{c^t_n}}.
\end{align*}
Define
\begin{align*}
v_C=\frac{1}{\sqrt{N}}\sum_{t=1}^N\ket{\psi_{c^t}}\quad\text{and}\quad\rho_C=\frac{1}{N}\sum_{s=1}^N\ket{\psi_{c^s}}\bra{\psi_{c^s}}.
\end{align*}
Put $\ket{\psi_C}={v_C}/{\|v_C\|}$.

We will choose $N$ and $C$ so that $\ket{\psi_C}\bra{\psi_C}$ and $\rho_C$ are hard to distinguish using low adaptive measurements.
That is to say, for every allowed measurement basis, the corresponding outcome distributions are similar.
The key is that the cross terms are small for every allowed measurement basis.
Consequently, any low adaptive algorithm that accepts $\ket{\psi_C}\bra{\psi_C}$ with high probability must also accept $\rho_C$ with the same probability, and hence cannot reliably certify $\ket{\psi_C}$.

We first show that $\|v_C\|^2$ is close to $1$ under suitable choice of $N$.

\begin{lemma}\label{lemma_19}
Let $N=2^{10^{-10}n}$ and sample $C=(c^1,\cdots,c^N)$ uniformly from $\prod_{k=1}^N\{1,2,3,4\}^n$.
If $n$ is large enough, then
\begin{align}\label{eq_202606131546}
\mathbb{P}\left(1-2^{-0.1n}\le\|v_C\|^2\le1+2^{-0.1n}\right)\ge0.9.
\end{align}
\end{lemma}
\begin{proof}
We first bound the probability that there exists a pair $s\neq t$ such that $d_{\text{H}}(c^s,c^t)\le0.3n$.
For fixed $s\neq t$,
\begin{align*}
d_{\text{H}}(c^s,c^t)\sim\mathrm{Binomial}\left(n,\frac{3}{4}\right).
\end{align*}
Hence, by the Chernoff bound,
\begin{align}\label{eq_202606111146}
\mathbb{P}\left(d_{\text{H}}(c^s,c^t)\le 0.3n\right)\le\sum_{k=0}^{0.3n}\binom{n}{k}\left(\frac{3}{4}\right)^{k}\left(\frac{1}{4}\right)^{n-k}\le e^{-nD(0.3\|\frac{3}{4})}\le2^{-0.64n}.
\end{align}
By the union bound, we have
\begin{align*}
\mathbb{P}\left(\exists s\neq t, d_{\text{H}}(c^s,c^t)\le 0.3n\right)&\le2^{2\cdot10^{-10}n-0.64n}\le2^{-0.6n}\le0.1
\end{align*}
for large $n$.
Thus,
\begin{align*}
\mathbb{P}\left(\forall s\neq t,d_{\text{H}}(c^s,c^t)> 0.3n\right)\ge1-0.1=0.9.
\end{align*}

Next, we estimate $\|v_C\|^2$.
For any $a\neq b$, we have
\begin{align*}
\left|\bra{\chi_{a}}\ket{\chi_{b}}\right|^2=\frac{1}{2}\left(1+r_a\cdot r_b\right)=\frac{1}{2}\left(1+\left(-\frac{1}{3}\right)\right)=\frac{1}{3}.
\end{align*}
Therefore, with probability at least $0.9$, we have
\begin{align*}
\left|\|v_C\|^2-1\right|
&=\left|\frac{1}{N}\sum_{s,t=1}^N\left(\bra{\chi_{c^s_1}}\otimes\cdots\otimes\bra{\chi_{c^s_n}}\right)\left(\ket{\chi_{c^t_1}}\otimes\cdots\otimes\ket{\chi_{c^t_n}}\right)-1\right|\\
&=\left|\frac{1}{N}\sum_{s,t=1}^N\bra{\chi_{c^s_1}}\ket{\chi_{c^t_1}}\cdots\bra{\chi_{c^s_n}}\ket{\chi_{c^t_n}}-1\right|\\
&=\frac{1}{N}\left|\sum_{s\neq t}\bra{\chi_{c^s_1}}\ket{\chi_{c^t_1}}\cdots\bra{\chi_{c^s_n}}\ket{\chi_{c^t_n}}\right|\\
&\le\frac{1}{N}\sum_{s\neq t}\left(\sqrt{\frac{1}{3}}\right)^{0.3n}=\frac{1}{N}N(N-1)\cdot\left(\sqrt{\frac{1}{3}}\right)^{0.3n}
\le N\cdot\left(\sqrt{\frac{1}{3}}\right)^{0.3n}\le2^{-0.1n}.
\end{align*}
This proves the lemma.
\end{proof}

From now on, suppose that $N=2^{10^{-10}n}$ and $C$ is sampled to satisfy \eqref{eq_202606131546}.
\begin{lemma}
Let $\{\ket{\varphi_x}\}_{x\in\{0,1\}^n}$ be an orthonormal basis for $(\mathbb{C}^2)^{\otimes n}$, and let $\mathcal{D}_1,\mathcal{D}_2$ be the outcome distributions by measuring $\ket{\psi_C}\bra{\psi_C}$ and $\rho_C$ in this basis, respectively.
That is,
\begin{align*}
\mathcal{D}_1(x)=\left|\bra{\varphi_x}\ket{\psi_C}\right|^2 \quad\text{and}\quad\mathcal{D}_2(x)=\bra{\varphi_x}\rho_C\ket{\varphi_x}.
\end{align*}
Then, for large $n$, we have
\begin{align*}
d_{\mathrm{TV}}\left(\mathcal{D}_1,\mathcal{D}_2\right)\le\frac{1}{N}\sum_{x}\left|\sum_{s\neq t}\bra{\varphi_x}\ket{\psi_{c^s}}\bra{\psi_{c^t}}\ket{\varphi_{x}}\right|+2^{-0.0001n}.
\end{align*}
\end{lemma}
\begin{proof}
By direct computations,
\begin{align}\label{eq_202606151049}
d_{\mathrm{TV}}\left(\mathcal{D}_1,\mathcal{D}_2\right)&=
\frac{1}{2}\sum_{x}\left|\mathcal{D}_1(x)-\mathcal{D}_2(x)\right|\notag\\
&=\frac{1}{2}\sum_{x}\left|\bra{\varphi_x}\ket{\psi_C}\bra{\psi_C}\ket{\varphi_x}-\bra{\varphi_x}\rho_C\ket{\varphi_x}\right|=\frac{1}{2}\sum_{x}\left|\bra{\varphi_x}\left(\ket{\psi_C}\bra{\psi_C}-\rho_C\right)\ket{\varphi_x}\right|.
\end{align}
Notice that
\begin{align*}
\|v_Cv_C^\dagger\|_1&=\tr\left(v_Cv_C^\dagger\right)=v_C^\dagger v_C=\|v_C\|^2.
\end{align*}
Thus, by Lemma~\ref{lemma_19}, we have
\begin{align}\label{eq_202606151045}
\|\ket{\psi_C}\bra{\psi_C}-v_Cv_C^\dagger\|_1&=\left\|\frac{1}{\|v_C\|^2}v_Cv_C^\dagger-v_Cv_C^\dagger\right\|_1=\left|\frac{1}{\|v_C\|^2}-1\right|\left\|v_Cv_C^\dagger\right\|_1\notag\\
&=\left|\frac{1}{\|v_C\|^2}-1\right|\|v_C\|^2=\left|1-\|v_C\|^2\right|\le2^{-0.1n}.
\end{align}
Moreover,
\begin{align*}
\rho_C=v_Cv_C^\dagger-\frac{1}{N}\sum_{s\neq t}\ket{\psi_{c^s}}\bra{\psi_{c^t}}.
\end{align*}
Therefore, we can write
\begin{align*}
\ket{\psi_C}\bra{\psi_C}-\rho_C=\frac{1}{N}\sum_{s\neq t}\ket{\psi_{c^s}}\bra{\psi_{c^t}}+E
\end{align*}
where $\|E\|_1\le2^{-0.1n}$ by \eqref{eq_202606151045}.
Continue \eqref{eq_202606151049},
\begin{align*}
d_{\mathrm{TV}}\left(\mathcal{D}_1,\mathcal{D}_2\right)&\le
\frac{1}{2}\left(\frac{1}{N}\sum_{x}\left|\sum_{s\neq t}\bra{\varphi_x}\ket{\psi_{c^s}}\bra{\psi_{c^t}}\ket{\varphi_x}\right|+\sum_{x}\left|\bra{\varphi_x}E\ket{\varphi_x}\right|\right)\\
&\le\frac{1}{N}\sum_{x}\left|\sum_{s\neq t}\bra{\varphi_x}\ket{\psi_{c^s}}\bra{\psi_{c^t}}\ket{\varphi_x}\right|+2^{-0.1n}\\
&\le\frac{1}{N}\sum_{x}\left|\sum_{s\neq t}\bra{\varphi_x}\ket{\psi_{c^s}}\bra{\psi_{c^t}}\ket{\varphi_x}\right|+2^{-0.0001n}.\\
\end{align*}
This proves the lemma.
\end{proof}

\begin{definition}
An orthonormal basis $\{\ket{\varphi_x}\}$ is said to be $\ell$-\textit{adaptive} if there exists $S\subset[n]$ with $|S|\le\ell$ such that
\begin{align*}
\ket{\varphi_x}=\bigotimes_{k\notin S}\ket{\varphi^k_{x_k}}\otimes\ket{\varphi^{x_{[n]-S}}_{x_S}}
\end{align*}
for each $x\in\{0,1\}^n$.
\end{definition}
This definition has the following operational interpretation.
The qubits with indices $k\notin S$ can be measured non-adaptively first, using the fixed bases $\{\ket{\varphi^k_0},\ket{\varphi^k_1}\}$.
If this measurement gives outcome $y=x_{[n]-S}$, then the remaining compatible basis becomes
\begin{align*}
\left\{\bigotimes_{k\notin S}\ket{\varphi^k_{y_k}}\otimes\ket{\varphi^y_z}\mid z\in\{0,1\}^{|S|}\right\}.
\end{align*}
Thus, after observing $y$, one measures the qubits in $S$ in the basis $\{\ket{\varphi^y_z}\mid z\in\{0,1\}^{|S|}\}$.
This shows that measurement in this basis can be performed adaptively.

We now show the most crucial estimation.
\begin{lemma}\label{lemma_21}
If $n$ is large enough, then we have for all $10^{-8}n$-adaptive bases $\{\ket{\varphi_x}\}$ that
\begin{align*}
\sum_{x}\left|\sum_{s\neq t}\bra{\varphi_x}\ket{\psi_{c^s}}\bra{\psi_{c^t}}\ket{\varphi_{x}}\right|\le5.
\end{align*}
\end{lemma}
Before proving Lemma~\ref{lemma_21}, we show a key obsevation.
A pair $(s,t)$ is \textit{bad} for an $\ell$-adaptive basis $\{\ket{\varphi_x}\}$ if for at least $10^{-6}n$ indices $k\in[n]-S$, one has $(c^s_k,c^t_k)=(0,b(\varphi^k)$.


\begin{claim}
Let $\{s_1,t_1\},\cdots,\{s_5,t_5\}$ be five distinct pairs in $[N]\times[N]$.
Then, with probability at least $1-2^{-10^{-5}n}$, at least one pair is bad for that $10^{-8}n$-adaptive basis.
\end{claim}
\begin{proof}
Let $G=(V=[N],E)$ be a graph such that $\{s_1,t_1\},\cdots,\{s_5,t_5\}\in E$.
We first show that there exists a labeling so that they form 
\begin{align*}
\{1,2\},\{1,3\},\{1,4\}
\end{align*}
in certain $k$.
We consider two cases.
\begin{enumerate}
\item
If $G$ contains a vertex $u$ with three edges from the five ones, then we can set
\begin{align*}
c^u_k=1,\,c^{v_1}_k=2,\,c^{v_2}_k=3,\,c^{v_3}_k=4.
\end{align*}
\item
If $G$ has maximum degree at most at most $2$, then $G$ is a disjoint union of paths and cycles.
Hence, among these five edges, we can choose either three disjoint edges or two adjacent edges together with one edge disjoint from them.
In both cases, we can assign a valid labeling.
\end{enumerate}
Since each label is choosing randomly from $[4]$, the probability that such pattern happens is at least $4^{-6}$.
Define a random variable $X_k$ such that $X_k=1$ if coordinate $k$ has the desired labeling, while $X_k=0$ if not.
Thus, we have
\begin{align*}
\mathbb{P}\left(X_k=1\right)\ge4^{-6}.
\end{align*}
Let $w=\sum_{k=1}^nX_k$.
By the Chernoff inequality, we have
\begin{align*}
\mathbb{P}\left(w\le\frac{4^{-6}n}{4}\right)\le e^{-\frac{(3/4)^2}{2}\cdot4^{-6}n}\le2^{-10^{-5}n}.
\end{align*}
Thus, by the pigeonhole principle, at least one pair is hit at least $4^{-6}n/20>10^{-5}n$ times.
Since $|S|\le10^{-8}n$, this pair is hit on at least
\begin{align*}
10^{-5}n-10^{-8}n\ge10^{-6}n
\end{align*}
coordinates in $S^c$.
Hence this pair is bad for the chosen basis.

Finally, because the basis is arbitrary, for every $10^{-8}n$-adaptive basis, at least one of the five pairs is bad for that basis.
This proves the claim.
\end{proof}

We now prove the key lemma.
\begin{proof}[Proof of Lemma \ref{lemma_21}]
By adaptivity, there is $S\subset[n]$ with $|S|\le10^{-8}n$.
Thus,
\begin{align*}
\sum_{x}\left|\sum_{s\neq t}\bra{\varphi_x}\ket{\psi_{c^s}}\bra{\psi_{c^t}}\ket{\varphi_{x}}\right|
&\le\sum_{x}\sum_{s\neq t}\left|\bra{\varphi_x}\ket{\psi_{c^s}}\bra{\psi_{c^t}}\ket{\varphi_{x}}\right|\\
&\le\sum_{s\neq t}\sum_{x}\prod_{k\notin S}\left|\bra{\varphi_{x_k}^k}\ket{\psi^s_{{c_k}}}\bra{\psi^t_{{c_k}}}\ket{\varphi^k_{x_{k}}}\right|\\
&=\sum_{s\neq t}\prod_{k\notin S}\left(\left|\bra{\varphi_{0}^k}\ket{\psi^s_{{c_k}}}\bra{\psi^t_{{c_k}}}\ket{\varphi^k_{0}}\right|+\left|\bra{\varphi_{1}^k}\ket{\psi^s_{{c_k}}}\bra{\psi^t_{{c_k}}}\ket{\varphi^k_{1}}\right|\right).
\end{align*}
By the union bound over all $N^{10}$ quintuples, the probability that $C$ is such that for every $10^{-8}n$-adaptive basis, at least one pair is bad for that basis is at least
\begin{align*}
1-2^{10^{-5}n}\cdot N^{10}\ge1-2^{-10^{-6}n}.
\end{align*}
Hence, at most four pairs are not bad.
Therefore,
\begin{align*}
\sum_{s\neq t}\prod_{k\notin S}\left(\left|\bra{\varphi_{0}^k}\ket{\psi^s_{{c_k}}}\bra{\psi^t_{{c_k}}}\ket{\varphi^k_{0}}\right|+\left|\bra{\varphi_{1}^k}\ket{\psi^s_{{c_k}}}\bra{\psi^t_{{c_k}}}\ket{\varphi^k_{1}}\right|\right)\le4+N^2\cdot\left(\sqrt{\frac{8}{9}}\right)^{-10^{-6}n}\le5
\end{align*}
for large $n$.
\end{proof}
